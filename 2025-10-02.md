        public void Move()
        {
            Console.WriteLine("Car 가 움직인다.");
        }
    }
    internal class Program
    {
        static void Main(string[] args)
        {
            {
                Car car = new Car();

                car.Move();
                Console.WriteLine();
                car = null;

                if (car != null)
                {
                    car.Move();
                }
                else
                {
                    Console.WriteLine("car 가 null 입니다");
                }
                Console.WriteLine();
                //Null Conditional operator
                car?.Move();

                car = new Car();

                car._weight = 10;

                // ?? : Null병합연산자
                int weight = car._weight ??= 0; // car._weight가 null이면 0, 아니면 그 값을 반환

                Console.WriteLine($"car._weight = {weight}");
                car._weight = null;

                Console.WriteLine($"car._weight = {weight}");

                Console.WriteLine();
                // Nullable

                int a;
                int? b = null; // 값타입인 a는 null 대입이 불가능

                Console.WriteLine($"b에 값이 {b.HasValue}");
                Console.WriteLine($"b != null {b != null}");
                Console.WriteLine();
                b = 3;
                Console.WriteLine($"b에 값이 {b.HasValue}");

            string str1 = "01x3455x90";
            int index = str1.IndexOf("x");  // 0부터 시작해서 x문자의 index위치를 리턴한다

            Console.WriteLine(index);

            index = str1.LastIndexOf("x");  // 뒤에서부터 앞쪽으로 이동하면서 x문자의 index위치를 리턴함

            Console.WriteLine(index);

            string str2 = "aaa (1234) cccc [5678] {ee}";
            char[] checkChar = { '{', '{', '[' };
            int endPosition = 0;

            while(true)
            {
                int startPosition = str2.IndexOfAny(checkChar, endPosition);
                // checkChar 배열에 저장되어있는 문자중에 아무거나 찾은 위치를 리턴함

                if (startPosition == -1) break; // 문자가 비교하는 문자열에 없는경우
                string startChar = str2.Substring(startPosition, 1); // 찾은 문자가 뭔지 확인

                char endChar = ' ';
                switch (startChar)
                {
                    case "{":
                        endChar = '}';
                        break;
                    case "[":
                        endChar = ']';
                        break;
                    case "(":
                        endChar = ')';
                        break;
                }
                startPosition++;
                endPosition = str2.IndexOf(endChar, startPosition); // endChar에 등록된 문자의 위치를 찾음
                int length = endPosition - startPosition; // 찾은 괄호사이에 있는 문자열의 길이를 계산
                Console.WriteLine(str2.Substring(startPosition, length)); // 괄호한에 문자열을 잘라서 출력

                
            }
            Console.WriteLine("\n ㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡㅡ\n");

            // ------------------------------------------
            // Remove() 삭제, Replace() 대치, Insert() 삽입
            string phoneNumber = "010-1234-5689";
            string result = phoneNumber.Remove(3, 4);
            Console.WriteLine(result);

            result = phoneNumber.Replace("-", "_");
            Console.WriteLine(result);

            result = phoneNumber.Insert(3, "(Call)");
            Console.WriteLine(result);
        }
    }
