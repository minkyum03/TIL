                goto Exit;   // 하향식분기

            //       명령어;

            for (int i = 0; i < 10; i++) // 단일 for문

                Console.WriteLine($"i = {i}");
            }

            for (int i = 0, j = 10; i < 10 && j > 0; i += 2, j -= 2)
            {
                Console.WriteLine($"i = {i}, j = {j}");
            }

            for (int i = 0; i < 10; i++) // 이중 for문
            {
                for (int j = 0; j < 10; j++)
                {
                    Console.WriteLine($"i = {i}, j = {j}");
                }
            }

            // while문은 반복횟수는 모르겠지만
            // while문의 조건식이 참인 동안에는 얼마든지 반복해도 괜찮을때.

                a -= 123.34f;

            }

            a = 12312341.1234f;

            do
            {
                a -= 123.234f;
                Console.WriteLine($"a = {a}");
