        static void BitPrint(int value)
        {
            int mask = 1;
            int bitValue = 0;

            for (int i = sizeof(int) * 8 - 1; i >= 0; i--)
            {
                mask = 1;
                mask <<= i;

                bitValue = value & mask;

                if (bitValue > 0)
                {
                    Console.Write("1");

                }
                else
                {
                    Console.Write("0");
                }
            }
        }
